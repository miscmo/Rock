198 bool checkGetAs(const MapType& m, const std::string& key, T& val, const T& def = T()) {
199  auto it = m.find(key);
200  if(it == m.end()) {
201  val = def;
202  return false;
203  }
204  try {
205  val = boost::lexical_cast<T>(it->second);
206  return true;
207  } catch (...) {
208  val = def;
209  }
210  return false;
211 }
