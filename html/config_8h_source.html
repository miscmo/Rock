482  RWMutexType::WriteLock lock(GetMutex());
483  auto it = GetDatas().find(name);
484  if(it != GetDatas().end()) {
485  auto tmp = std::dynamic_pointer_cast<ConfigVar<T> >(it->second);
486  if(tmp) {
487  ROCK_LOG_INFO(ROCK_LOG_ROOT()) << "Lookup name=" << name << " exists";
488  return tmp;
489  } else {
490  ROCK_LOG_ERROR(ROCK_LOG_ROOT()) << "Lookup name=" << name << " exists but type not "
491  << TypeToName<T>() << " real_type=" << it->second->getTypeName()
492  << " " << it->second->toString();
493  return nullptr;
494  }
495  }
496 
497  if(name.find_first_not_of("abcdefghikjlmnopqrstuvwxyz._012345678")
498  != std::string::npos) {
499  ROCK_LOG_ERROR(ROCK_LOG_ROOT()) << "Lookup name invalid " << name;
500  throw std::invalid_argument(name);
501  }
502 
503  typename ConfigVar<T>::ptr v(new ConfigVar<T>(name, default_value, description));
504  GetDatas()[name] = v;
505  return v;
